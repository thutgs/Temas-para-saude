Relatório Temas para a Saúde

options(scipen = 999)
library(pacman)
p_load(readxl, openxlsx, tidyverse, dplyr, reshape2, kableExtra, moments, glue)
banco <- read_rds("temas.rds")
indicadores <- read_rds("indicadores.rds")
quadro <- function(x){
  banco %>% 
    group_by(ano) %>% 
     summarise(média = mean({{x}}),
            `desvio padrão` = sd({{x}}),
            máximo = max({{x}}),
            `1_quartil` = quantile({{x}}, .25),
            mediana = median({{x}}),
            `3_quartil` = quantile({{x}}, .75),
            assimetria = skewness({{x}}),
            curtose = kurtosis({{x}}))
}
